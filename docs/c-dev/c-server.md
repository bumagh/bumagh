# lwan
lwan.ws
https://github.com/lpereira/lwan
## clone
~$ git clone git://github.com/lpereira/lwan
~$ cd lwan
## build
~/lwan$ mkdir build
~/lwan$ cd build
~/lwan/build$ cmake .. -DCMAKE_BUILD_TYPE=Release
~/lwan/build$ cmake .. -DCMAKE_BUILD_TYPE=RelWithDebInfo
~/lwan/build$ cmake .. -DCMAKE_BUILD_TYPE=Debug
~/lwan/build$ make

sudo make install
