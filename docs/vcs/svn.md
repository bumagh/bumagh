# git vs svn
git init = svn create
git clone = svn co
git add = svn add 
git commit = svn commit
git pull = svn update
git status = svn status
git branch = svn cp
git branch -d = svn rm
git merge = svn merge
git differ = svn diff
git checkout = svn update -r
git checkout tag = svn switch tag
git checkout branch = svn switch branch
git checkout - path = svn revert path
git rm path = svn rm path
git push = svn update
